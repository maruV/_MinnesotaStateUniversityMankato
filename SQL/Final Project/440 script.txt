CREATE TABLE COMPANY(
COMPANY_ID NUMBER NOT NULL PRIMARY KEY,
COMPANY_NAME VARCHAR(20),
COMPANY_COUNTRY VARCHAR(20),
COMPANY_STATE VARCHAR(20),
COMPANY_CITY VARCHAR(20),
COMPANY_STREET VARCHAR(30),
COMPANY_PHONE NUMBER(10));

CREATE TABLE UK(
UK_ADDRESS VARCHAR(20),
UK_CITY VARCHAR(20),
COMPANY_ID NUMBER NOT NULL REFERENCES COMPANY(COMPANY_ID));

CREATE TABLE US(
US_ADDRESS VARCHAR(20),
US_CITY VARCHAR(20),
US_STATE VARCHAR(10),
COMPANY_ID NUMBER NOT NULL REFERENCES COMPANY(COMPANY_ID));

CREATE TABLE DEPARTMENT(
DEPT_ID NUMBER NOT NULL PRIMARY KEY,
DEPARTMENT_NAME VARCHAR(20));

CREATE TABLE SUPERVISOR(
SUPERV_ID NUMBER NOT NULL PRIMARY KEY,
SUPERVISOR_NAME VARCHAR(20),
BONUS NUMBER);

CREATE TABLE EMPLOYEE(
EMP_ID NUMBER NOT NULL PRIMARY KEY,
EMP_F_NAME VARCHAR(20),
EMP_L_NAME VARCHAR(20),
EMP_ADDRESS VARCHAR(20),
EMP_PHONE NUMBER,
EMP_EMAIL VARCHAR(20),
EMP_HOUR_RATE NUMBER(20)),
COMPANY_ID NUMBER NOT NULL REFERENCES COMPANY(COMPANY_ID),
SUPERV_ID NUMBER NOT NULL REFERENCES SUPERVISOR(SUPERV_ID),
DEPT_ID NUMBER NOT NULL REFERENCES DEPARTMENT(DEPT_ID);

CREATE TABLE TASK(
TASK_ID NUMBER NOT NULL PRIMARY KEY,
TASK_START_DATE DATE,
TASK_DUE_DATE DATE);

CREATE TABLE PROJECT(
PROJECT_ID NUMBER NOT NULL PRIMARY KEY,
PROJECT_TITLE VARCHAR(20),
PROJECT_DESCRIPTION VARCHAR(20),
PROJECT_START_DATE DATE,
PROJECT_END_DATE DATE);

CREATE TABLE TIMECARD(
PAY_PER_HOUR VARCHAR(20),
PROJECT_MONTH VARCHAR(20),
TOTAL_HOURS NUMBER,
EMP_ID NUMBER NOT NULL REFERENCES EMPLOYEE(EMP_ID),
TASK_ID NUMBER NOT NULL REFERENCES TASK(TASK_ID),
PROJECT_ID NUMBER NOT NULL REFERENCES PROJECT(PROJECT_ID));

CREATE TABLE CLIENT(
CLIENT_ID NUMBER PRIMARY KEY,
CLI_LEGAL_FIRST_NAME VARCHAR(20),
CLI_LEGAL_LAST_NAME VARCHAR(20),
CLI_COMMON_FIRST_NAME VARCHAR(20),
CLI_COMMON_LAST_NAME VARCHAR(20),
CLI_GENDER VARCHAR(10),
CLI_BILL_COUNTRY VARCHAR(20),
CLI_BILL_CITY VARCHAR(20),
CLI_BILL_ZIP_CODE VARCHAR(20),
CLI_BILL_STREET VARCHAR(40),
CLI_PHONE NUMBER,
CLI_EMAIL VARCHAR(20));

CREATE TABLE CONTRACTS(
CONTRACT_ID NUMBER PRIMARY KEY,
CLIENT_ID NUMBER NOT NULL REFERENCES CLIENT(CLIENT_ID),
PROJECT_ID NUMBER NOT NULL REFERENCES PROJECT(PROJECT_ID),
CONTRACT_FIXED_PRICES NUMBER,
CONTRACT_MAX_HOURS NUMBER,
CONTRACTS_HOURLY_PRICE NUMBER);

CREATE TABLE SALES(
SALES_ID NUMBER PRIMARY KEY,
SALES_DATE DATE,
EMP_ID NUMBER NOT NULL REFERENCES EMPLOYEE(EMP_ID),
CONTRACT_ID NUMBER NOT NULL REFERENCES CONTRACTS(CONTRACT_ID));

CREATE TABLE TRANSACTION(
TRANSACTION_ID NUMBER PRIMARY KEY,
TRASACTION_AMOUNT NUMBER,
PAYMENT_METHOD VARCHAR(20),
TRASACTION_DATE DATE,
SALES_ID NUMBER NOT NULL REFERENCES SALES(SALES_ID));
